#set up nginx conf file for signalR websockets

server{
	listen 80;
	server_name **your vm ip address or host name**;

	location / {
		proxy_pass http://**your url from the blazor app**;
		proxy_set_header Upgrade $http_upgrade;
		proxy_set_header Connection "Upgrade";
		proxy_cache off;
		proxy_http_version 1.1 
		proxy_buffering off;
		proxy_read_timeout 100s; //VERY IMPORTANT TO EXTEND LISTENING TIME FOR WEBSOCKETS
		proxy_set_header Host $host;
		proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
		proxy_set_header X-Forwarded-Proto $scheme;
	
	}

}
